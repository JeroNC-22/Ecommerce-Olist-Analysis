SQL — Query destacada fase_5_sellers_y_delivery.sql
Ver código completo 
SELECT
    o.order_id,
    CASE
        WHEN o.order_delivered_customer_date <= o.order_estimated_delivery_date
        THEN 'A tiempo'
        ELSE 'Tardía'
    END                          AS tipo_entrega,
    AVG(r.review_score)         AS score_promedio
FROM olist_orders_dataset o
INNER JOIN olist_order_reviews_dataset r ON o.order_id = r.order_id
WHERE o.order_status = 'delivered'
GROUP BY tipo_entrega;

-- Resultado: entregas a tiempo → 4.29 ⭐  |  tardías → 2.27 ⭐
-- Una entrega tardía destruye la satisfacción del cliente.
